39  subroutine compgetu ( n, NNZ, a, ja, ia, ao, jao, iao )
40  !
41  ! GETU extracts the upper triangular part of a matrix.
42  !
43  ! Discussion:
44  !
45  ! The routine writes the result ao, jao, iao.
46  !
47  ! The routine is in place in that ao, jao, iao can be the same
48  ! as a, ja, ia if desired.
49  !
50  ! The diagonal element is the last element in each row.
51  ! i.e. in a(ia(i+1)-1 )
52  ! ao, jao, iao may be the same as a, ja, ia on entry --
53  ! in which case getu will overwrite the result on a, ja, ia.
54  !
55  ! Modified:
56  !
57  ! 07 January 2004
58  !
59  ! Author:
60  !
61  ! Youcef Saad
62  !
63  ! Parameters:
64  !
65  ! Input, integer ( kind = 4 ) N, the order of the matrix.
66  !
67  ! Input, real A(*), integer ( kind = 4 ) JA(*), IA(N+1),
68  ! the matrix in CSR Compressed Sparse Row format.
69  !
70  ! Input, real AO(*), JAO(*), IAO(N+1), the upper triangular
71  ! part of the input matrix, in CSR Compressed Sparse Row format.
72  !
73 
74  ! INPUT
75  integer, intent(in) :: n
76  integer, intent(in) :: NNZ
77  complex(kind=dp), dimension(:), intent(in) :: a
78  integer, dimension(:), intent(in) :: ia
79  integer, dimension(:), intent(in) :: ja
80 
81 
82  ! OUTPUT
83  complex(kind=dp), dimension(:), intent(inout) :: ao
84  integer, dimension(:), intent(inout) :: iao
85  integer, dimension(:), intent(inout) :: jao
86 
87  ! LOCAL variables
88  integer :: i
89  integer :: k
90  integer :: kdiag
91  integer :: kfirst
92  integer :: ko
93  complex(kind=dp) :: t
94 
95  !-------------------------------------------------------------------
96  ! check array dimensions
97  if (size(ao) .ne. nnz) &
98  call write_error_message(log_unit, "compgetu: ao has wrong size")
99  if (size(iao) .ne. n+1) &
100  call write_error_message(log_unit, "compgetu: iao has wrong size")
101  if (size(jao) .ne. nnz) &
102  call write_error_message(log_unit, "compgetu: jao has wrong size")
103 
104 
105  ko = 0
106 
107  do i = 1, n
108 
109  kfirst = ko + 1
110  kdiag = 0
111 
112  do k = ia(i), ia(i+1)-1
113 
114  if ( i <= ja(k) ) then
115  ko = ko + 1
116  ao(ko) = a(k)
117  jao(ko) = ja(k)
118  if ( ja(k) == i ) then
119  kdiag = ko
120  end if
121  end if
122 
123  end do
124  !
125  ! Exchange.
126  !
127  if ( kdiag /= 0 .and. kdiag /= kfirst ) then
128 
129  t = ao(kdiag)
130  ao(kdiag) = ao(kfirst)
131  ao(kfirst) = t
132 
133  k = jao(kdiag)
134  jao(kdiag) = jao(kfirst)
135  jao(kfirst) = k
136 
137  end if
138 
139  iao(i) = kfirst
140 
141  end do
142  !
143  ! Redefine IAO(N+1).
144  !
145  iao(n+1) = ko + 1
146 
147  end subroutine compgetu

155  subroutine compcoicsr (n,nnz,job,a,ja,ia)
156 
157  ! INPUT
158  integer, intent(in) :: n, nnz, job
159 
160  ! IN/OUTPUT
161  complex(kind=dp), dimension(:), intent(inout) :: a
162  integer, dimension(:), intent(inout) :: ja, ia
163 
164  ! LOCAL variables
165  integer, dimension(n+1) :: iwk
166  complex(kind=dp) :: t, tnext
167  logical :: values, run_init_loop
168  integer :: i, inext, init, j, jnext, k, ipos
169 
170  !-------------------------------------------------------------------
171  ! IN-PLACE coo-csr conversion routine.
172  !-------------------------------------------------------------------
173  ! this subroutine converts a matrix stored in coordinate format into
174  ! the csr format. The conversion is done in place in that the arrays
175  ! a,ja,ia of the result are overwritten onto the original arrays.
176  !-------------------------------------------------------------------
177  ! on entry:
178  !---------
179  ! n = integer. row dimension of A.
180  ! nnz = integer. number of nonzero elements in A.
181  ! job = integer. Job indicator. when job=1, the real values in a are
182  ! filled. Otherwise a is not touched and the structure of the
183  ! array only (i.e. ja, ia) is obtained.
184  ! a = real array of size nnz (number of nonzero elements in A)
185  ! containing the nonzero elements
186  ! ja = integer array of length nnz containing the column positions
187  ! of the corresponding elements in a.
188  ! ia = integer array of length max(nnz,n+1) containing the row
189  ! positions of the corresponding elements in a.
190  ! iwk = integer work array of length n+1
191  ! on return:
192  !----------
193  ! a
194  ! ja
195  ! ia = contains the compressed sparse row data structure for the
196  ! resulting matrix.
197  ! Note:
198  !-------
199  ! the entries of the output matrix are not sorted (the column
200  ! indices in each are not in increasing order) use coocsr
201  ! if you want them sorted. Note also that ia has to have at
202  ! least n+1 locations
203  !-------------------------------------------------------------------
204  ! Coded by Y. Saad, Sep. 26 1989
205  !-------------------------------------------------------------------
206  !-------------------------------------------------------------------
207  t = (0.0_dp, 0.0_dp)
208  tnext = (0.0_dp, 0.0_dp)
209  values = (job .eq. 1)
210 
211 
212  ! find pointer array for resulting matrix.
213  do i = 1,n+1
214  iwk(i) = 0
215  end do
216  do k = 1,nnz
217  i = ia(k)
218  iwk(i+1) = iwk(i+1)+1
219  end do
220  !-------------------------------------------------------------------
221  iwk(1) = 1
222  do i = 2,n
223  iwk(i) = iwk(i-1) + iwk(i)
224  end do
225  !
226  ! loop for a cycle in chasing process.
227  !
228  init = 1
229  k = 0
230  chasing_loop: do
231  if (values) t = a(init)
232  i = ia(init)
233  j = ja(init)
234  ia(init) = -1
235 
236  !----------------------------------------------------------------
237  inner_loop: do while (k .lt. nnz)
238  k = k+1
239  ! current row number is i. determine where to go.
240  ipos = iwk(i)
241  ! save the chased element.
242  if (values) tnext = a(ipos)
243  inext = ia(ipos)
244  jnext = ja(ipos)
245  ! then occupy its location.
246  if (values) a(ipos) = t
247  ja(ipos) = j
248  ! update pointer information for next element to come in row i
249  iwk(i) = ipos+1
250  ! determine next element to be chased,
251  if (ia(ipos) .lt. 0) then
252  run_init_loop = .true.
253  exit inner_loop
254  else
255  run_init_loop = .false.
256  end if
257  t = tnext
258  i = inext
259  j = jnext
260  ia(ipos) = -1
261  end do inner_loop
262 
263  select case (run_init_loop)
264  case (.false.)
265  exit chasing_loop
266  case (.true.)
267  init_loop: do
268  init = init+1
269  if (init .gt. nnz) exit chasing_loop
270  if (ia(init) .ge. 0) exit init_loop
271  end do init_loop
272  end select
273  ! restart chasing --
274  end do chasing_loop
275 
276  do i = 1,n
277  ia(i+1) = iwk(i)
278  end do
279  ia(1) = 1
280 
281  end subroutine compcoicsr

290  subroutine csort (n, values, a, ja, ia)
291 
292  ! INPUT
293  integer, intent(in) :: n
294  logical, intent(in) :: values
295 
296  ! IN/OUTPUT
297  complex(kind=dp), dimension(:), intent(inout) :: a
298  integer, dimension(:), intent(inout) :: ja, ia
299 
300  ! LOCAL variables
301  integer :: row, i, k, j
302  complex(kind=dp) :: rj
303 
304  !---------------------------------------------------------------------
305  ! This routine sorts the elements of a matrix (stored in Compressed
306  ! Sparse Row Format in increasing order of their column indices within
307  ! each row. It uses insertion sort.
308  !---------------------------------------------------------------------
309  ! on entry:
310  !---------
311  ! n = the row dimension of the matrix
312  ! a = the matrix A in compressed sparse row format.
313  ! ja = the array of column indices of the elements in array a.
314  ! ia = the array of pointers to the rows.
315  ! values= logical indicating whether or not the real values a(*) must
316  ! also be permuted. if (.not. values) then the array a is not
317  ! touched by csort and can be a dummy array.
318  !
319  ! on return:
320  !----------
321  ! the matrix stored in the structure a, ja, ia is permuted in such a
322  ! way that the column indices are in increasing order within each row.
323  ! iwork(1:nnz) contains the permutation used to rearrange the elements
324  !---------------------------------------------------------------------
325  ! Y. Saad - recoded Dec. 20th - 2017 -
326  !---------------------------------------------------------------------
327 
328  rj = (0.0_dp,0.0_dp)
329 
330  ! for each row
331 
332  do row=1, n
333 
334  ! scan row and do an insertion sort
335 
336  do k=ia(row)+1, ia(row+1)-1
337  j = ja(k)
338  if (values) rj = a(k)
339  i = k-1;
340  do while ((i .ge. ia(row)) .and. (j < ja(i)) )
341  ja(i+1) = ja(i)
342  if (values) a(i+1) = a(i)
343  i = i-1
344  if (i .eq. 0) exit
345  enddo
346  ja(i+1) = j
347  if (values) a(i+1) = rj
348  enddo
349  enddo
350  return
351 
352  end subroutine csort

360  subroutine csumdup (Aj, Ap, Ax)
361 
362  ! Sum together duplicate column entries in each row of CSR matrix A
363  ! The column indicies within each row must be in sorted order.
364  ! Explicit zeros are retained.
365  ! Ap, Aj, and Ax will be modified *inplace*
366  !---------------------------------------------------------------------
367 
368  ! INPUT/Output
369  integer, intent(inout) :: Ap(:), Aj(:)
370  complex(kind=dp), intent(inout) :: Ax(:)
371 
372  !Local variables
373  integer :: nnz, r1, r2, i, j, jj
374  complex(kind=dp) :: x
375 
376  !---------------------------------------------------------------------
377 
378 
379  nnz = 1
380  r2 = 1
381  do i = 1, size(ap) - 1
382  r1 = r2
383  r2 = ap(i+1)
384  jj = r1
385  do while (jj < r2)
386  j = aj(jj)
387  x = ax(jj)
388  jj = jj + 1
389  do while (jj < r2)
390  if (aj(jj) == j) then
391  x = x + ax(jj)
392  jj = jj + 1
393  else
394  exit
395  end if
396  end do
397  aj(nnz) = j
398  ax(nnz) = x
399  nnz = nnz + 1
400  end do
401  ap(i+1) = nnz
402  end do
403 
404  end subroutine csumdup

412  subroutine compcsrcoo (nrow,job,nzmax,a,ja,ia,nnz,ao,ir,jc,ierr)
413 
414  ! INPUT
415  integer, intent(in) :: nrow, job, nzmax
416  complex(kind=dp), dimension(:), intent(inout) :: a
417  integer, dimension(:), intent(inout) :: ja, ia
418  ! OUTPUT
419  complex(kind=dp), dimension(:), intent(out) :: ao
420  integer, dimension(:), intent(out) :: ir, jc
421  integer, intent(out) :: nnz, ierr
422  ! LOCAL variables
423  integer :: i, k, k1, k2
424 
425  !
426  ! on entry:
427  !---------
428  ! nrow = dimension of the matrix.
429  ! job = integer serving as a job indicator.
430  ! if job = 1 fill in only the array ir, ignore jc, and ao.
431  ! if job = 2 fill in ir, and jc but not ao
432  ! if job = 3 fill in everything.
433  ! The reason why these options are provided is that on return
434  ! ao and jc are the same as a, ja. So when job = 3, a and ja are
435  ! simply copied into ao, jc. When job=2, only jc and ir are
436  ! returned. With job=1 only the array ir is returned. Moreover,
437  ! the algorithm is in place:
438  ! call csrcoo (nrow,1,nzmax,a,ja,ia,nnz,a,ia,ja,ierr)
439  ! will write the output matrix in coordinate format on a, ja,ia.
440  !
441  ! a,
442  ! ja,
443  ! ia = matrix in compressed sparse row format.
444  ! nzmax = length of space available in ao, ir, jc.
445  ! the code will stop immediatly if the number of
446  ! nonzero elements found in input matrix exceeds nzmax.
447  !
448  ! on return:
449  !-----------
450  ! ao, ir, jc = matrix in coordinate format.
451  !
452  ! nnz = number of nonzero elements in matrix.
453  ! ierr = integer error indicator.
454  ! ierr .eq. 0 means normal retur
455  ! ierr .eq. 1 means that the the code stopped
456  ! because there was no space in ao, ir, jc
457  ! (according to the value of nzmax).
458  !
459  ! NOTES: 1)This routine is PARTIALLY in place: csrcoo can be called
460  ! with ao being the same array as as a, and jc the same
461  ! array as ja. but ir CANNOT be the same as ia.
462  ! 2) note the order in the output arrays,
463  !-------------------------------------------------------------------
464  ierr = 0
465  nnz = ia(nrow+1)-1
466  ! print *,'nnz in iAgcsr_o', nnz
467  if (nnz .gt. nzmax) then
468  ierr = 1
469  return
470  end if
471  !-------------------------------------------------------------------
472  select case (job)
473  case (1)
474  !
475  ! copy backward to allow for in-place processing.
476  !
477  do i = nrow,1,-1
478  k1 = ia(i+1)-1
479  k2 = ia(i)
480  do k=k1,k2,-1
481  ir(k) = i
482  end do
483  end do
484  case (2)
485  do k = 1,nnz
486  jc(k) = ja(k)
487  end do
488  case (3)
489  do k = 1,nnz
490  ao(k) = a(k)
491  end do
492  end select
493 
494  end subroutine compcsrcoo
