553  curl_w2 = curl_w2 + &
554  cmplx(0.0_dp, 1.0_dp/(w * mu(elem2)), kind=dp) &
555  * cmplx(curl_n, kind=dp) * wg(elem2_ed(l))
556 
557  end do
558 
559 
560  ! Calculate face jumps J with dot product for current face
561  !factor = cmplx(0.0_dp, w, kind=dp)
562  fjj_tmp_s = dot_product(cmplx(face_normal,kind=dp), &
563  cmplx(s1 * (1.0_dp/rho(elem1)) - &
564  s2 * (1.0_dp/rho(elem2)), &
565  kind=dp))
566  fjj_tmp_w = dot_product(cmplx(face_normal,kind=dp), &
567  cmplx(w1 * (1.0_dp/rho(elem1)) - &
568  w2 * (1.0_dp/rho(elem2)), &
569  kind=dp))
570 
571 
572  ! Calculate face jumps H with cross product for current face
573  fjh_tmp_s = cross_product(cmplx(face_normal,kind=dp), &
574  cmplx(curl_s1 - curl_s2, kind=dp))
575  fjh_tmp_w = cross_product(cmplx(face_normal,kind=dp), &
576  cmplx(curl_w1 - curl_w2, kind=dp))
577 
578 
579  ! absolute values of electric fields at faces
580  abs_s1 = sqrt((abs(s1(1)))**2.0_dp + &
581  (abs(s1(2)))**2.0_dp + &
582  (abs(s1(3)))**2.0_dp)
583  abs_s2 = sqrt((abs(s2(1)))**2.0_dp + &
584  (abs(s2(2)))**2.0_dp + &
585  (abs(s2(3)))**2.0_dp)
586  abs_w1 = sqrt((abs(w1(1)))**2.0_dp + &
587  (abs(w1(2)))**2.0_dp + &
588  (abs(w1(3)))**2.0_dp)
589  abs_w2 = sqrt((abs(w2(1)))**2.0_dp + &
590  (abs(w2(2)))**2.0_dp + &
591  (abs(w2(3)))**2.0_dp)
592 
593  ! absolute values of magnetic fields at faces
594  abs_curl_s1 = sqrt((abs(curl_s1(1)))**2.0_dp + &
595  (abs(curl_s1(2)))**2.0_dp + &
596  (abs(curl_s1(3)))**2.0_dp)
597  abs_curl_s2 = sqrt((abs(curl_s2(1)))**2.0_dp + &
598  (abs(curl_s2(2)))**2.0_dp + &
599  (abs(curl_s2(3)))**2.0_dp)
600  abs_curl_w1 = sqrt((abs(curl_w1(1)))**2.0_dp + &
601  (abs(curl_w1(2)))**2.0_dp + &
602  (abs(curl_w1(3)))**2.0_dp)
603  abs_curl_w2 = sqrt((abs(curl_w2(1)))**2.0_dp + &
604  (abs(curl_w2(2)))**2.0_dp + &
605  (abs(curl_w2(3)))**2.0_dp)
606 
607  ! Weighting factors for relative field amplitudes:
608  be = ((abs((abs_s1 + abs_s2)/2)) + 10e-9)**2.0_dp
609  bw = ((abs((abs_w1 + abs_w2)/2)) + 10e-9)**2.0_dp
610  ce = ((abs((abs_curl_s1 + abs_curl_s2)/2)) + 10e-9)**2.0_dp
611  cw = ((abs((abs_curl_w1 + abs_curl_w2)/2)) + 10e-9)**2.0_dp
612 
613 
614  ! Sum the weighted norm of the face jumps up over all 4
615  ! neighbouring faces of current element i scaled with 0.5
616  ! divide by abs((abs_S1**2 + abs_S2**2)/2 + 10E-7**2) to get
617  ! a relative face jump (relative to field amplitude)
618  fjj_s(i) = fjj_s(i) + &
619  (0.5_dp * abs(fjj_tmp_s * fjj_tmp_s)/ abs(be)) &
620  * area
621  fjj_w(i) = fjj_w(i) + &
622  (0.5_dp * abs(fjj_tmp_w * fjj_tmp_w)/ abs(bw)) &
623  * area
624  fjh_s(i) = fjh_s(i) + &
625  (0.5_dp * &
626  abs((dot_product(fjh_tmp_s, fjh_tmp_s)))/abs(ce))&
627  * area
628  fjh_w(i) = fjh_w(i) + &
629  (0.5_dp * &
630  abs((dot_product(fjh_tmp_w, fjh_tmp_w)))/abs(cw))&
631  * area
632 
633  ! end if of elem2 is existing
634  end if
635 
636  ! end loop over 4 neighbouring elements
637  end do
638 
639  ! end loop over elements
640  end do
641  !$OMP END PARALLEL DO
642  !---------------------------------------------------------------------
643  end subroutine calculate_face_jumps

651  subroutine compute_elemental_error_estimates (errorEst_method, M, &

655  eta_L, Sum_eta_L, &

683  eta_l = 0.0_dp
684  eta_s = 0.0_dp

687  ! test outputs
688  ! print *, 'Max primal elemental residual:', maxval(res_s)
689  ! print *, 'Max dual elemental residual:', maxval(res_w)
690  ! print *, 'Max primal face jump J:', maxval(fjJ_s)

702  eta_w(i) = sqrt(abs(res_w(i)))

712  eta_l(i) = eta_s(i) * eta_w(i)
713  end do
714 
715  case (3)
716  ! Element error estimator with residuals, face jumps J and H
717  ! Start loop over elements:
718  do i = 1,m
719  eta_s(i) = sqrt(abs(res_s(i)) + abs(fjj_s(i)) + abs(fjh_s(i)))
720  eta_w(i) = sqrt(abs(res_w(i)) + abs(fjj_w(i)) + abs(fjh_w(i)))
721  eta_l(i) = eta_s(i) * eta_w(i)
722  end do
723 
724  case (4)
725  ! Element error estimator with face jumps J
726  ! Start loop over elements:

729  eta_w(i) = sqrt(abs(fjj_w(i)))
730  eta_l(i) = eta_s(i) * eta_w(i)

737  eta_s(i) = sqrt(abs(fjh_s(i)))
738  eta_w(i) = sqrt(abs(fjh_w(i)))
739  eta_l(i) = eta_s(i) * eta_w(i)
740  end do
741 
742  case (6)
743  ! Element error estimator with face jumps J and face jumps H
744  ! Start loop over elements:
745  do i = 1,m
746  eta_s(i) = sqrt(abs(fjj_s(i)) + abs(fjh_s(i)))
747  eta_w(i) = sqrt(abs(fjj_w(i)) + abs(fjh_w(i)))
748  eta_l(i) = eta_s(i) * eta_w(i)
749  end do
750 
751  end select
752 
753 
754  ! Sum of element error estimates over all elements
755  sum_eta_l = 0.0_dp
756  sum_eta_l = sum(eta_l)
757  !-------------------------------------------------------------------
758  end subroutine compute_elemental_error_estimates

768  subroutine calculate_face_normal(p1, p2, p3, face_normal)
769 
770  ! INPUT
771  real(kind=dp), dimension(3), intent(in) :: p1, p2, p3
772 
773  ! OUTPUT
774  real (kind=dp), dimension(3), intent(out) :: face_normal
775 
776  ! LOCAL variables
777  integer :: i
778  real (kind=dp) :: l
779  real (kind=dp), dimension(3) :: u, v
780 
781  !--------------------------------------------------------------------
782  ! Calculate vector v: from p1 to p2
783  ! and u: from p1 to p3
784 
785  do i = 1,3
786  u(i) = p2(i) - p1(i)
787  v(i) = p3(i) - p1(i)
788  end do
789 
790  ! Cross product of u and v is the face normal
791  face_normal(1) = u(2) * v(3) - u(3) * v(2)
792  face_normal(2) = u(3) * v(1) - u(1) * v(3)
793  face_normal(3) = u(1) * v(2) - u(2) * v(1)
794

796  l = sqrt(face_normal(1)**2.0_dp + &
797  face_normal(2)**2.0_dp + &
798  face_normal(3)**2.0_dp)
799 
800  ! Scale face normal by its length to get unit length
801  do i = 1,3
802  face_normal(i) = face_normal(i) * (1.0_dp/l)
803  end do
804

806  end subroutine calculate_face_normal

814  subroutine triangle_area(p1, p2, p3, area)
815 
816  ! INPUT
817  real(kind=dp), dimension(3), intent(in) :: p1, p2, p3
818 
819  ! OUTPUT
820  real (kind=dp), intent(out) :: area
821 
822  ! LOCAL variables
823  real (kind=dp) :: a, b, c, s
824 
825  !-------------------------------------------------------------------
826  ! Calculate edge lengths for triangle:
827  ! a: p1 -> p2, b: p2 -> p3, c: p3 -> p1
828  a = sqrt( (p2(1) - p1(1))**2.0_dp + &
829  (p2(2) - p1(2))**2.0_dp + &
830  (p2(3) - p1(3))**2.0_dp)
831  b = sqrt( (p3(1) - p2(1))**2.0_dp + &
832  (p3(2) - p2(2))**2.0_dp + &
833  (p3(3) - p2(3))**2.0_dp)

835  (p1(2) - p3(2))**2.0_dp + &
836  (p1(3) - p3(3))**2.0_dp)
837 
838  ! Semiperimeter of triangle
839  s = 0.5_dp * (a + b + c)
840 
841  ! Area (using Heron's formula:
842  ! http://www.mathopenref.com/heronsformula.html
843  area = sqrt(s * (s - a) * (s - b) * (s - c))
844

846  end subroutine triangle_area
